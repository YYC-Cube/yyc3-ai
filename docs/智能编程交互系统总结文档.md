### 场景2: 渐进式代码生成
\`\`\`typescript
import { multiTurnDialogOptimizer } from '@/lib/multi-turn-dialog-optimizer'

// 启动渐进式生成
const plan = await multiTurnDialogOptimizer.startProgressiveGeneration(
  '创建一个博客管理系统'
)

// 处理用户响应
const result = await multiTurnDialogOptimizer.processStepResponse(
  plan.id,
  '用户模型包含: id, name, email, avatar',
  'confirm'
)

// 获取进度
const progress = multiTurnDialogOptimizer.getProgress(plan.id)
console.log(`进度: ${progress.percentage}% (${progress.completed}/${progress.total})`)
\`\`\`
